from ruth.core import RuthAssistant
from ruth.ears import RuthEars
import sys

def main():
    # 1. Instanciar Cerebro y Oído
    app = RuthAssistant()
    ears = RuthEars()
    
    app.introduce_self()
    app.speak("Estoy lista para escucharte. Di 'reporte', 'descargas' o 'adiós'.")
    
    # 2. EL BUCLE DE VIDA (Service Loop)
    while True:
        # Ruth se queda bloqueada aquí esperando a que hables
        text = ears.listen()
        
        if not text:
            continue

        # --- LÓGICA DE COMANDOS ---
        
        if "reporte" in text or "mantenimiento" in text:
            app.create_log_entry()
            
        elif "descargas" in text or "archivos" in text:
            app.inspect_downloads()
            
        elif "notepad" in text and "abrir" in text:
            app.execute_app("notepad")
            
        elif "adiós" in text or "apagar" in text or "terminar" in text:
            app.speak("Desconectando sistemas. Hasta luego, Inge.")
            break # Rompe el bucle y termina el programa
            
        else:
            # Si no entendió
            # app.speak("No entendí la orden.") # Opcional: a veces es mejor que calle si no entiende
            print(f"Comando no reconocido: {text}")

if __name__ == "__main__":
    main()
